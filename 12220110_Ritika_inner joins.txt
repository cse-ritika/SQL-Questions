1 - NAME OF THE EMPLOYEE AND HIS LOCATION OF ALL THE EMPLOYEES. 
SQL> SELECT e.ename, d.loc FROM emp e JOIN dept d ON e.deptno=d.deptno;

2 - WAQTD DNAME AND SALARY FOR ALL THE EMPLOYEE WORKING IN ACCOUNTING. 
SQL> SELECT d.dname, e.sal FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE d.dname='ACCOUNTING';

3 - WAQTD DNAME AND ANNUAL SALARY FOR ALL EMPLOYEES WHOSE SALARY IS MORE THAN 2340. 
SQL> SELECT d.dname, e.sal*12 AS annual_salary FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE e.sal>2340;

4 - WAQTD ENAME AND DNAME FOR EMPLOYEES HAVING CHARACTER 'A' IN THEIR DNAME. 
SQL> SELECT e.ename, d.dname FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE d.dname LIKE '%A%';

5 - WAQTD ENAME AND DNAME FOR ALL THE EMPLOYEES WORKING AS SALESMAN. 
SQL> SELECT e.ename, d.dname FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE e.job='SALESMAN';

6 - WAQTD DNAME AND JOB FOR ALL THE EMPLOYEES WHOSE JOB AND DNAME STARTS WITH CHARACTER 'S'. 
SQL> SELECT d.dname, e.job FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE e.job LIKE 'S%' AND d.dname LIKE 'S%';

7 - WAQTD DNAME AND MGR NO FOR EMPLOYEES REPORTING TO 7839. 
SQL> SELECT d.dname, e.mgr FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE e.mgr=7839;

8 - WAQTD DNAME AND HIREDATE FOR EMPLOYEES HIRED AFTER 83 INTO ACCOUNTING OR RESEARCH DEPT. 
SQL> SELECT d.dname, e.hiredate FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE e.hiredate>'31-DEC-83' AND d.dname IN ('ACCOUNTING','RESEARCH');

9 - WAQTD ENAME AND DNAME OF THE EMPLOYEES WHO ARE GETTING COMM IN DEPT 10 OR 30. 
SQL> SELECT e.ename, d.dname FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE e.comm IS NOT NULL AND e.deptno IN (10,30);

10 - WAQTD DNAME AND EMPNO FOR ALL THE EMPLOYEES WHOSE EMPNO ARE (7839,7902) AND ARE WORKING IN LOC NEW YORK. 
SQL> SELECT d.dname, e.empno FROM emp e JOIN dept d ON e.deptno=d.deptno WHERE e.empno IN (7839,7902) AND d.loc='NEW YORK';
