1 - WAQTD THE EMPLOYEE DETAILS WHO ARE EARNING MORE THAN 2000.
SQL> select * from emp where SAL>2000;

2 - WAQTD THE EMPLOYEE NAME, SAL, COMM, JOB WHO ARE HIRED AFTER '28-SEP-81'.
SQL> select ENAME,SAL,COMM,JOB from emp where HIREDATE > '28-SEP-81' ;

3 - WAQTD THE DEPARTMENT NAME WHICH LOCATED IN DALLAS.
SQL> select DNAME from dept where LOC = 'DALLAS';

4 - WAQTD THE SAL, ANNUAL SALARY WHO ARE EARNING ANNUAL SALARY MORE THAN 20000.
SQL> select sal, sal*12 as Annual_Salary from emp where (sal*12) > 20000;

5 - WAQTD THE EMPNO, DEPTNO, COMM WHO ARE WORKING AS SALESMAN.
SQL> select empno,deptno,comm from emp where job = 'SALESMAN';

6 - LIST DOWN THE EMPLOYEE DETAILS WHO ARE WORKING AS A MANAGER.
SQL> select * from emp where JOB = 'MANAGER';

7 - DISPLAY THE EMPLOYEE SALARY AND 30% MONTHLY HIKE.
SQL> select SAL, SAL*1.30 as MONTHLY_HIKE from emp;

8 - DISPLAY THE EMPLOYEE SALARY, ANNUAL SALARY ALONG WITH 30% ANNUAL HIKE.
SQL> select SAL, SAL*12 as ANNUAL_SALARY, (SAL*12)*1.30 as ANNUAL_HIKE from emp;

9 - DISPLAY THE EMPLOYEE SALARY, ANNUAL SALARY WHO AND ALL PRESENT IN DEPTNO 30.
SQL> select SAL, SAL*12 as ANNUAL_SALARY from emp where DEPTNO = 30;

10 - DISPLAY THE EMPLOYEE SALARY, HALF SALARY OF THE EMPLOYEES WHO ARE HIRED AFTER '17-DEC-81'.
SQL> select SAL, SAL/2 as HALF_SALARY from emp where HIREDATE > '17-DEC-81';

11 - DISPLAY THE EMPLOYEE SALARY WITH HIKE OF 10%, ANNUAL SALARY WITH HIKE OF 50%.
SQL> select SAL*1.10 as SALARY_HIKE_10, (SAL*12)*1.50 as ANNUAL_HIKE_50 from emp;

12 - WAQTD THE EMPLOYEE DETAILS WHO JOB IS NOT A CLERK.
SQL> select * from emp where JOB <> 'CLERK';

13 - LIST DOWN EMPLOYEE DETAILS WHO ARE WORKING IN DEPTNO 20.
SQL> select * from emp where DEPTNO = 20;

14 - SHOW ME THE EMPLOYEE SAL AND COMM WITH THE ADDITION OF 20 IF THEY ARE BELONGS TO MANAGER JOB.
SQL> select SAL+20 as SALARY, COMM+20 as COMMISSION from emp where JOB = 'MANAGER';

15 - LIST DOWN THE ALL THE EMPLOYEE DETAILS IF THEY ARE EARNING COMMISSION AS ZERO.
SQL> select * from emp where COMM = 0;
