===============================
	GROUP BY CLAUSE
===============================

1 - WAQTD NUMBER OF EMPLOYEES WORKING IN EACH DEPARTMENT EXCEPT PRESIDENT.
SQL> SELECT deptno, COUNT(*) FROM emp WHERE job != 'PRESIDENT' GROUP BY deptno;

2 - WAQTD TOTAL SALARY NEEDED TO PAY ALL THE EMPLOYEES IN EACH JOB.
SQL> SELECT job, SUM(sal) FROM emp GROUP BY job;

3 - WAQTD NUMBER OF EMPLOYEES WORKING AS MANAGER IN EACH DEPARTMENT.
SQL> SELECT deptno, COUNT(*) FROM emp WHERE job = 'MANAGER' GROUP BY deptno;

4 - WAQTD AVG SALARY NEEDED TO PAY ALL THE EMPLOYEES IN EACH DEPARTMENT EXCLUDING DEPTNO 20.
SQL> SELECT deptno, AVG(sal) FROM emp WHERE deptno != 20 GROUP BY deptno;

5 - WAQTD NUMBER OF EMPLOYEES HAVING CHARACTER 'A' IN THEIR NAMES IN EACH JOB.
SQL> SELECT job, COUNT(*) FROM emp WHERE ename LIKE '%A%' GROUP BY job;

6 - WAQTD NUMBER OF EMPLOYEES AND AVG SALARY OF EMPLOYEES WHOSE SALARY IS GREATER THAN 2000 IN EACH DEPARTMENT.
SQL> SELECT deptno, COUNT(*), AVG(sal) FROM emp WHERE sal > 2000 GROUP BY deptno;

7 - WAQTD TOTAL SALARY NEEDED TO PAY AND NUMBER OF SALESMAN IN EACH DEPARTMENT.
SQL> SELECT deptno, COUNT(*), SUM(sal) FROM emp WHERE job = 'SALESMAN' GROUP BY deptno;

8 - WAQTD NUMBER OF EMPLOYEES WITH THEIR MAXIMUM SALARIES IN EACH JOB.
SQL> SELECT job, COUNT(*), MAX(sal) FROM emp GROUP BY job;

9 - WAQTD MAXIMUM SALARY GIVEN TO AN EMPLOYEE WORKING IN EACH DEPARTMENT.
SQL> SELECT deptno, MAX(sal) FROM emp GROUP BY deptno;

10 - WAQTD NUMBER OF TIMES THE SALARIES ARE PRESENT IN EMP TABLE.
SQL> SELECT sal, COUNT(*) FROM emp GROUP BY sal;

11 - WAQTD NUMBER OF EMPLOYEES HIRED ON THE SAME DAY INTO THE SAME DEPARTMENT.
SQL> SELECT hiredate, deptno, COUNT(*) FROM emp GROUP BY hiredate, deptno HAVING COUNT(*) > 1;

12 - WAQTD NUMBER OF EMPLOYEES GETTING THE SAME SALARY IN THE SAME DEPARTMENT.
SQL> SELECT sal, deptno, COUNT(*) FROM emp GROUP BY sal, deptno HAVING COUNT(*) > 1;

13 - WAQTD NUMBER OF EMPLOYEES HAVING THE SAME NAME IN THE SAME DEPARTMENT.
SQL> SELECT ename, deptno, COUNT(*) FROM emp GROUP BY ename, deptno HAVING COUNT(*) > 1;

14 - WAQTD NUMBER OF EMPLOYEES GETTING THE SAME SALARY IN THE SAME DEPARTMENT AND HAVING THE SAME NAME.
SQL> SELECT ename, sal, deptno, COUNT(*) FROM emp GROUP BY ename, sal, deptno HAVING COUNT(*) > 1;

15 - WAQTD NUMBER OF EMPLOYEES GETTING THE SAME SALARY IN THE SAME DEPARTMENT AND HIRED ON THE SAME DAY.
SQL> SELECT sal, deptno, hiredate, COUNT(*) FROM emp GROUP BY sal, deptno, hiredate HAVING COUNT(*) > 1;
